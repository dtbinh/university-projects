 
% TW244:  Oplos van modelprobleem f(x) = exp(x)-4*x met Newton se metode

F  = inline('exp(x)-4*x');      % Definieer funksie
DF = inline('exp(x)-4');        % Definieer afgeleide


toler = 1e-10;                  % Spesifiseer toleransie
del = 1;                        % Inisialiseer delta                                     
x   = 2;                        % Aanvanklike skatting
X   = x;                        % Stoor benaderings in X

while abs(del) > toler;         % Begin iterasie
    del = F(x)/DF(x);           % Bereken delta
    x = x - del                 % Newton se metode
    X = [X; x];                 % Voeg nuwe x by lys van oues
end
